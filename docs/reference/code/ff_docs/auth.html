120    def check_permission(
121        self, session: UserSession, check: PermissionCheck
122    ) -> PermissionResult:
123        """Check if user has required permissions for a resource."""
124        try:
125            # Define permission patterns
126            permission_patterns = {
127                "/health": [],  # Public endpoint
128                "/repos/config": [],  # Public endpoint
129                "/repos/": ["docs:read"],
130                "/repos/discover": ["docs:read"],
131                "/repos/enroll": ["repos:manage"],
132                "/repos/unenroll": ["repos:manage"],
133                "/repos/enroll-all": ["repos:manage"],
134            }
135
136            # Check for admin paths
137            if check.resource.startswith("/admin"):
138                required_permissions = ["docs:admin"]
139            else:
140                # Get required permissions for resource
141                required_permissions = permission_patterns.get(
142                    check.resource, ["docs:read"]
143                )
144
145            # Check if user has any of the required permissions
146            user_permissions = set(session.permissions)
147            required_perms_set = set(required_permissions)
148
149            if not required_permissions or required_perms_set.intersection(
150                user_permissions
151            ):
152                return PermissionResult(
153                    allowed=True,
154                    reason="User has required permissions",
155                    required_permissions=required_permissions,
156                )
157            return PermissionResult(
158                allowed=False,
159                reason=f"Missing required permissions: {required_permissions}",
160                required_permissions=required_permissions,
161            )
162
163        except Exception:
164            logger.exception("Error checking permissions")
165            return PermissionResult(
166                allowed=False,
167                reason="Permission check failed",
168                required_permissions=[],
169            )

 26    async def authenticate_user(self, github_token: str) -> UserSession | None:
 27        """Authenticate user with GitHub token and create session."""
 28        try:
 29            # Get user information from GitHub
 30            user = await self._get_github_user(github_token)
 31            if not user:
 32                return None
 33
 34            # Get user's team memberships
 35            teams = await self._get_user_teams(github_token, user.username)
 36
 37            # Generate session
 38            session_id = str(uuid.uuid4())
 39            expires_at = datetime.now(UTC) + timedelta(
 40                minutes=self.settings.auth.access_token_expire_minutes
 41            )
 42
 43            # Create user session
 44            session = UserSession(
 45                user=user,
 46                teams=teams,
 47                permissions=self._calculate_permissions(teams),
 48                session_id=session_id,
 49                expires_at=expires_at,
 50            )
 51
 52        except Exception:
 53            logger.exception("GitHub authentication failed")
 54            return None
 55        else:
 56            logger.info("Successfully authenticated user: %s", user.username)
 57            return session

 59    async def _get_github_user(self, token: str) -> User | None:
 60        """Get user information from GitHub API."""
 61        headers = {
 62            "Authorization": f"Bearer {token}",
 63            "Accept": "application/vnd.github+json",
 64            "X-GitHub-Api-Version": "2022-11-28",
 65        }
 66
 67        try:
 68            async with httpx.AsyncClient() as client:
 69                response = await client.get(
 70                    f"{self.base_url}/user", headers=headers
 71                )
 72                response.raise_for_status()
 73
 74                user_data = response.json()
 75
 76                # Update last login
 77                now = datetime.now(UTC)
 78
 79                return User(
 80                    username=user_data["login"],
 81                    email=user_data.get("email", ""),
 82                    full_name=user_data.get("name"),
 83                    avatar_url=user_data.get("avatar_url"),
 84                    github_id=user_data.get("id"),
 85                    is_active=True,
 86                    created_at=now,
 87                    last_login=now,
 88                )
 89
 90        except httpx.HTTPStatusError:
 91            logger.exception("GitHub API error getting user")
 92            return None
 93        except Exception:
 94            logger.exception("Error getting GitHub user")
 95            return None

 97    async def _get_user_teams(
 98        self, token: str, username: str
 99    ) -> list[GitHubTeam]:
100        """Get user's team memberships from GitHub."""
101        headers = {
102            "Authorization": f"Bearer {token}",
103            "Accept": "application/vnd.github+json",
104            "X-GitHub-Api-Version": "2022-11-28",
105        }
106
107        teams: list[GitHubTeam] = []
108        org_name = self.settings.github.org
109
110        if not org_name:
111            logger.warning("No GitHub organization configured")
112            return teams
113
114        try:
115            async with httpx.AsyncClient() as client:
116                # Get user's teams in the organization
117                response = await client.get(
118                    f"{self.base_url}/orgs/{org_name}/teams", headers=headers
119                )
120                response.raise_for_status()
121
122                org_teams = response.json()
123
124                # Check membership for each team
125                for team in org_teams:
126                    team_slug = team["slug"]
127                    member_response = await client.get(
128                        f"{self.base_url}/orgs/{org_name}/teams/{team_slug}/memberships/{username}",
129                        headers=headers,
130                    )
131
132                    if member_response.status_code == httpx.codes.OK:
133                        membership = member_response.json()
134                        if membership.get("state") == "active":
135                            teams.append(
136                                GitHubTeam(
137                                    org=org_name,
138                                    team=team_slug,
139                                    role=membership.get("role", "member"),
140                                )
141                            )
142
143                logger.info(
144                    "Found %d team memberships for user: %s",
145                    len(teams),
146                    username,
147                )
148                return teams
149
150        except httpx.HTTPStatusError as e:
151            if e.response.status_code == httpx.codes.NOT_FOUND:
152                logger.info(
153                    "User %s not found in organization %s", username, org_name
154                )
155            else:
156                logger.exception("GitHub API error getting teams")
157            return teams
158        except Exception:
159            logger.exception("Error getting user teams")
160            return teams
161
162    def _calculate_permissions(self, teams: list[GitHubTeam]) -> list[str]:
163        """Calculate user permissions based on team memberships."""
164        permissions = set()
165
166        # Base permissions for authenticated users
167        permissions.add("docs:read")
168
169        # Team-based permissions
170        team_permissions = {
171            "admin-team": [
172                "docs:read",
173                "docs:write",
174                "docs:admin",
175                "repos:manage",
176                "users:manage",
177            ],
178            "platform-team": [
179                "docs:read",
180                "docs:write",
181                "repos:manage",
182                "system:monitor",
183            ],
184            "docs-team": ["docs:read", "docs:write", "repos:view"],
185            "backend-team": ["docs:read", "backend:access"],
186            "frontend-team": ["docs:read", "frontend:access"],
187            "mobile-team": ["docs:read", "mobile:access"],
188        }
189
190        for team in teams:
191            team_perms = team_permissions.get(team.team, ["docs:read"])
192            permissions.update(team_perms)
193
194        return list(permissions)
