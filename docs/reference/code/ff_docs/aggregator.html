 93                    "direction": "desc",
 94                }

 96                try:

100                        params=params,
101                        timeout=30.0,
102                    )

104
105                    repos_data = response.json()
106                    if not repos_data:
107                        break
108
109                    for repo_data in repos_data:
110                        repo_info = self._parse_repository_data(repo_data)

116                    logger.exception(

166
167                    if response.status_code == 200:  # noqa: PLR2004
168                        content_data = response.json()
169

187                                repo_info.name,

190
191                except httpx.HTTPStatusError:
192                    # File/directory not found, continue checking

301            batch = tasks[i : i + batch_size]
302            results = await asyncio.gather(*batch, return_exceptions=True)
303
304            for result in results:
305                if isinstance(result, Exception):
306                    logger.error("Error checking repository: %s", result)
307                    continue
308
309                # Type guard to ensure result is RepositoryInfo
310                if isinstance(result, RepositoryInfo) and result.has_docs:
311                    doc_repositories.append(result)
312
313            # Brief pause between batches
314            if i + batch_size < len(tasks):
315                await asyncio.sleep(1)
316
317        logger.info(
318            "Found %d repositories with documentation", len(doc_repositories)
319        )
320        return doc_repositories

 38        Args:
 39            config: Enrollment configuration
 40        """

 44        self.aggregator = RepositoryAggregator()

 47    async def enroll_repository(
 48        self,
 49        repository: str | RepositoryInfo,
 50        section: str | None = None,
 51    ) -> bool:
 52        """Enroll a single repository in the documentation system.
 53
 54        Args:
 55            repository: Repository name or RepositoryInfo object
 56            section: Navigation section name (defaults to repository name)
 57
 58        Returns:
 59            True if enrollment successful
 60        """
 61        if isinstance(repository, str):
 62            # Discover repository information
 63            repositories = (
 64                await self.aggregator.discover_documentation_repositories()
 65            )
 66            repo_info = next(
 67                (r for r in repositories if r.name == repository), None
 68            )
 69            if not repo_info:
 70                logger.error("Repository not found: %s", repository)
 71                return False
 72        else:
 73            repo_info = repository
 74
 75        if not repo_info.has_docs:
 76            logger.warning(
 77                "Repository has no documentation: %s", repo_info.name
 78            )
 79            return False
 80
 81        # Load current MkDocs configuration
 82        mkdocs_config = self._load_mkdocs_config()
 83
 84        # Add repository to navigation
 85        section_name = section or repo_info.name.title()
 86        import_url = self._build_import_url(repo_info)
 87
 88        nav_entry = {section_name: f"!import {import_url}"}
 89
 90        # Find Projects section or create it
 91        nav = mkdocs_config.get("nav", [])
 92        projects_section = None
 93
 94        for item in nav:
 95            if isinstance(item, dict) and "Projects" in item:
 96                projects_section = item["Projects"]
 97                break
 98
 99        if projects_section is None:
100            # Create Projects section
101            projects_section = [{"Overview": "projects/index.md"}]
102            nav.append({"Projects": projects_section})
103            mkdocs_config["nav"] = nav
104
105        # Add repository to Projects section
106        projects_section.append(nav_entry)
107
108        # Save updated configuration
109        if self.config.backup_config:
110            self._backup_config()
111
112        self._save_mkdocs_config(mkdocs_config)
113
114        logger.info("Successfully enrolled repository: %s", repo_info.name)
115        return True

118        self,
119        org: str | None = None,
120        exclude: list[str] | None = None,
121    ) -> dict[str, bool]:
122        """Enroll all repositories with documentation from organization.
123
124        Args:
125            org: Organization name (defaults to configured org)
126            exclude: List of repository names to exclude
127
128        Returns:
129            Dictionary mapping repository names to enrollment success
130        """
131        exclude = exclude or []
132        repositories = (
133            await self.aggregator.discover_documentation_repositories(org)
134        )
135
136        # Filter out excluded repositories
137        repositories = [
138            repo for repo in repositories if repo.name not in exclude
139        ]
140
141        results = {}
142        for repo in repositories:
143            try:
144                success = await self.enroll_repository(repo)
145                results[repo.name] = success
146            except Exception:
147                logger.exception("Failed to enroll %s", repo.name)
148                results[repo.name] = False

238    def _load_mkdocs_config(self) -> dict[str, Any]:
239        """Load MkDocs configuration file.
240
241        Returns:
242            Configuration dictionary
243        """
244        config_path = self.config.target_mkdocs_config
245
246        if not config_path.exists():
247            logger.error("MkDocs config not found: %s", config_path)
248            msg = f"MkDocs config not found: {config_path}"
249            raise FileNotFoundError(msg)
250
251        # Create a custom YAML loader that handles MkDocs-specific tags
252        class MkDocsLoader(yaml.SafeLoader):
253            pass
254
255        def construct_python_name(
256            loader: Any,  # noqa: ARG001, ANN401
257            node: Any,  # noqa: ANN401
258        ) -> str:
259            """Handle !!python/name: tags by returning a placeholder."""
260            return f"!!python/name:{node.value}"
261
262        # Register constructors for specific MkDocs Python name tags
263        MkDocsLoader.add_constructor(
264            "tag:yaml.org,2002:python/name:material.extensions.emoji.twemoji",
265            construct_python_name,
266        )
267        MkDocsLoader.add_constructor(
268            "tag:yaml.org,2002:python/name:material.extensions.emoji.to_svg",
269            construct_python_name,
270        )
271        MkDocsLoader.add_constructor(
272            "tag:yaml.org,2002:python/name:pymdownx.superfences.fence_code_format",
273            construct_python_name,
274        )
275        MkDocsLoader.add_constructor(
276            "tag:yaml.org,2002:python/name:pymdownx.superfences.fence_div_format",
277            construct_python_name,
278        )
279
280        # Add a fallback constructor for any remaining python tags
281        MkDocsLoader.add_multi_constructor(  # type: ignore[no-untyped-call]
282            "tag:yaml.org,2002:python/",
283            lambda loader, suffix, node: f"!!python/{suffix}:{node.value}",  # noqa: ARG005
284        )
285
286        # Also register for the short form without tag prefix
287        MkDocsLoader.add_multi_constructor(  # type: ignore[no-untyped-call]
288            "!!python/", construct_python_name
289        )
290
291        with config_path.open(encoding="utf-8") as f:
292            config = yaml.load(f, Loader=MkDocsLoader)  # noqa: S506
293
294        return config or {}
